RTL-SDR dongle setup:
	# Update your Raspberry Pi package lists so you have the newest versions
	sudo apt update

	# Install the tools needed to build the V4 driver from source 
	# git lets you download source code
	# cmake and build-essential allow you to compile programs
	# libusb and pkg-config let the driver talk to the USB dongle
	sudo apt install -y git cmake build-essential libusb-1.0-0-dev pkg-config

	# Remove the old rtl-sdr package that comes with Raspberry Pi OS
	# The built-in package is outdated and does NOT fully support the V4 dongle
	sudo apt purge -y rtl-sdr

	# Disable the Linux kernel's built-in TV tuner driver
	# Without this step, the kernel grabs the dongle and blocks rtl_test/rtl_fm
	echo "blacklist dvb_usb_rtl28xxu" | sudo tee /etc/modprobe.d/blacklist-rtl.conf

	# Reboot the Pi so the blacklist takes effect and the dongle comes up clean
	sudo reboot

	# Check that the Pi sees the RTL-SDR Blog V4 device on USB after reboot
	lsusb | grep -i Realtek

Update the driver for hydrogen line:
	# Go to your home directory
	cd ~

	# Remove any old copy of the V4 source so you start clean
	rm -rf rtl-sdr-blog

	# Download the official RTL-SDR Blog V4 driver source code
	git clone https://github.com/rtlsdrblog/rtl-sdr-blog

	# Enter the source directory
	cd rtl-sdr-blog

	# Create a build directory
	mkdir build
	cd build

	# Configure the build
	# DETACH_KERNEL_DRIVER=ON lets the RTL app take control even if the kernel tries
	cmake .. -DDETACH_KERNEL_DRIVER=ON

	# Compile the driver
	make

	# Install the driver into the system
	sudo make install

	# Update the library cache so the system sees the new driver
	sudo ldconfig

	# Test the dongle to confirm V4 driver is active
	# Expected output includes:
	# "RTL-SDR Blog V4 Detected"
	# "Found Rafael Micro R828D tuner"
	
	rtl_test -t

	# 1. Connect your NooElec hydrogen-line mesh dish feed to the SAWbird+ H1 RF IN port.
	#    The H1 "input" faces the sky. The "output" goes toward the dongle.

	# 3. The dongle goes into your Raspberry Pi USB port.

	# 4. To power the SAWbird+ H1:
	#    The RTL-SDR Blog V4 provides bias tee power internally.
	#    So you DO NOT plug anything into the SAWbird USB port.
	#    You simply run: rtl_biast -b 1

	# At this point your hardware chain is:
	# Dish → SAWbird+ H1 → RTL-SDR Blog V4 → Raspberry Pi
	# -----------------------------------------------------------

Test the equipment indoors
	# Turn on the bias-tee to power the SAWbird H1 LNA
	rtl_biast -b 1
	# Expected output:
	#   Found 1 device(s):
	#     0:  RTLSDRBlog, Blog V4, SN: 00000001
	#   Using device 0: Generic RTL2832U OEM
	#   Found Rafael Micro R828D tuner
	#   RTL-SDR Blog V4 Detected

	# Run a hydrogen-line sweep around 1420 MHz

	rtl_power -f 1415M:1425M:1k -g 40 test.csv
	# Expected behavior:
	#   Number of frequency hops: 4
	#   Dongle bandwidth: 2500000Hz
	#   Logged FFT bins: 16384
	#   Reporting every 10 seconds
	#   Tuner gain set to 40.20 dB.
	#   Exact sample rate is ~2500000 Hz
	#   (Press Ctrl+C to stop after at least 20-30 seconds.)
	# After stopping:
	#   A file named test.csv appears in your home directory.
	#   This file contains the hydrogen-line power spectrum.

	# Confirm the file was created
	ls -lh test.csv
	# Expected:
	#   Something like: -rw-r--r-- 1 benschwem benschwem 700K <timestamp> test.csv
